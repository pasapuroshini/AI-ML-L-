## Write a program to check whether a substring is present a string.
```
import java.util.*;
public class Main{
    public  static int Checker(String str,String check)
    {
         int n=str.length();
        int m=check.length();

        for(int i=0;i<=n-m;i++)
        {
            int j=0;
            for(j=0;j<m;j++)
            {
                if(str.charAt(i+j)!=check.charAt(j))
                {
                    break;
                }
            }


            if(j==m)
            {
                return i;
            }
        }

        return -1;
    }
    public static void main(String[] args){
        Scanner sc= new Scanner(System.in);
        String str= sc.nextLine();
        String check= sc.nextLine();
        int startindex= Checker(str,check);
        System.out.print(startindex);
      
       
    }
}
```

```
#include <stdio.h>

int Checker(const char *str, const char *check) {
    int n = strlen(str);
    int m = strlen(check);

    for (int i = 0; i <= n - m; i++) {
        int j = 0;
        for (j = 0; j < m; j++) {
            if (str[i + j] != check[j]) {
                break;
            }
        }
        if (j == m) {
            return i;
        }
    }
    return -1;
}
```
